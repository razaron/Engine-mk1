225  inline void Graph<V, E, G>::breadthFirstTraversal(std::size_t origin)
226  {
227  // Return if origin has no adjacent vertices
228  if (!(*this)[origin].adjacencyList.size())
229  return;
230 
231  std::list<Vertex<V, E>> openList;
232  std::list<Vertex<V, E>> closedList;
233 
234  //add this vertex to openQueue
235  openList.push_back((*this)[origin]);
236 
237  //Process open queue
238  while (!openList.empty())
239  {
240  // Pop a vertex from the openList
241  auto v = *openList.begin();
242  openList.pop_front();
243 
244  // If the vertex is already in the closed list, go to the next loop
245  if (std::find(closedList.begin(), closedList.end(), (*this)[v.id]) != closedList.end())
246  continue;
247  else
248  closedList.push_back((*this)[v.id]);
249 
250  // Run the vertices discovery function
251  if (vertexFuncs[(*this)[v.id].state])
252  vertexFuncs[(*this)[v.id].state]((*this)[v.id], *this);
253 
254  if ((*this)[v.id].state == State::WHITE)
255  (*this)[v.id].state = State::GREY;
256 
257  // Loop through adjacent edges
258  for (auto &e : (*this)[v.id].adjacencyList)
259  {
260  // Run the edges discovery function
261  if (edgeFuncs[e.state])
262  edgeFuncs[e.state](e, *this);
263 
264  if (e.state == State::WHITE)
265  e.state = State::GREY;
266 
267  openList.push_back((*this)[e.target]);
268  }
269  }
270  }

427  inline void Graph<V, E, G>::branchHelper(std::size_t origin)
428  {
429  // Run the vertices discovery function
430  if (vertexFuncs[(*this)[origin].state])
431  vertexFuncs[(*this)[origin].state]((*this)[origin], *this);
432 
433  // recurse on the first non-grey child vertex
434  for (auto &e : (*this)[origin].adjacencyList)
435  {
436  if ((*this)[e.target].state != State::GREY)
437  {
438  // Run the edges discovery function
439  if (edgeFuncs[e.state])
440  edgeFuncs[e.state](e, *this);
441 
442  branchHelper(e.target);
443  break;
444  }
445  }
446 
447  bool unvisited = false;
448  for (auto &e : (*this)[origin].adjacencyList)
449  {
450  if ((*this)[e.target].state != State::GREY)
451  unvisited = true;
452  }
453 
454  if (!unvisited)
455  (*this)[origin].state = State::GREY;
456  }
