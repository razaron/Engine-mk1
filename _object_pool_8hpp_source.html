231  inline void Pool<S>::defragment()
232  {
233  std::scoped_lock lk{ _mutex };
234 
235  using Buffer = std::array<std::byte, S>;
236 
237  // TODO ??? return `count` to log objects moved or get rid of `count`
238  int count = 0;
239  // Loop throught allocated objects
240  for (auto & [ hdl, ptr ] : _map)
241  {
242  // If allocated object is located after first free position
243  if (getIndex(ptr) > _firstFreeLine->index)
244  {
245  count++;
246

259  template <std::size_t S>
260  inline void Pool<S>::shrink()
261  {
262  std::scoped_lock lk{ _mutex };
263 
264  if (_firstFreeLine == nullptr)
265  return;
266 
267  std::vector<LineType *> freePtrs{ _firstFreeLine };
268 
269  std::size_t lastPos = _pages.size() * OBJECT_POOL_PAGE_LENGTH - 1;
270 
271  // loop through all free lines
272  while (freePtrs.back()->next != nullptr)
273  {
274  freePtrs.push_back(freePtrs.back()->next);
275  }
276 
277  // return early if not enough free lines
278  if (freePtrs.size() < OBJECT_POOL_PAGE_LENGTH)
279  return;
280 
281  lastPos++;
282  size_t pos = freePtrs.size();
283  size_t toDelete = 0;
284  while (pos > 0)
285  {
286  pos -= OBJECT_POOL_PAGE_LENGTH;
287 
288  if (freePtrs[pos]->index == (lastPos -= OBJECT_POOL_PAGE_LENGTH))
289  toDelete++;
290  else
291  break;
292  }
293 
294  // If _firstFreeLine is in an unwanted page, set it to nullptr
295  if (_firstFreeLine->index >= (_pages.size() - toDelete) * OBJECT_POOL_PAGE_LENGTH)
296  _firstFreeLine = nullptr;
297 
298  // Erase unwanted pages
299  auto it = _pages.begin();
300  std::advance(it, _pages.size() - toDelete);
301 
302  _pages.erase(it, _pages.end());
303  }

307  inline void Pool<S>::addPage()
308  {
309  auto page = new PageType;
310 
311  // Initialize the pages lines with pointers pointing to the next free line
312  auto i = 0u;
313  for (; i < OBJECT_POOL_PAGE_LENGTH - 1; i++)
314  {
315  page->data()[i].next = &page->data()[i + 1];
316  page->data()[i].index = _pages.size() * OBJECT_POOL_PAGE_LENGTH + i;
317  }
318 
319  // Last line points to nullptr
320  page->data()[i].next = nullptr;
321 
322  // If first page, set _freePtr to the new pages first free line
323  if (_firstFreeLine == nullptr)
324  _firstFreeLine = &page->data()[0];
325  // Else set the last pages last line to the new pages first line
326  else
327  _pages.back()->data()[i].next = &page->data()[0];
328

396  inline void Pool<S>::eraseImpl(const Handle &handle, bool destruct)
397  {
398  if (destruct)
399  std::destroy_at(getImpl<T>(handle));
400 
401  LineType *ptrToRemove = getImpl<LineType>(handle);
402 
403  // If the object being removed is located BEFORE the first free position
404  if (getIndex(ptrToRemove) < getIndex(_firstFreeLine))
405  {
406  ptrToRemove->next = _firstFreeLine;
407  ptrToRemove->index = getIndex(ptrToRemove);
408 
409  _firstFreeLine = ptrToRemove;
410  }
411  // If the object being removed is located AFTER the first free position
412  else
413  {
414  LineType *prevFree = nullptr;
415  LineType *nextFree = _firstFreeLine;
416 
417  std::size_t pos = getIndex(ptrToRemove);
418 
419  // Loop through free positions until ptrToRemove is inbetween prevFree and nextFree
420  while (getIndex(nextFree) < pos)
421  {
422  prevFree = nextFree;
423  nextFree = nextFree->next;
424  }
425 
426  // Set ptrToRemove to be inbetween ptrPrevFree and ptrNextFree
427  prevFree->next = ptrToRemove;
428  ptrToRemove->next = nextFree;
429  ptrToRemove->index = pos;
430  }
